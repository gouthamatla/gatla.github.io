Joint_analysis_SNpGenePairs_pval = {}

npvalues = []
bpvalues = []
with open("/project/jferrer/goutham/eQTLs/splicingQTLs/leaveOutAnalysis/Joint/HI_LG_OXf_PISA_filt_perind.counts.gz.qqnorm_all_permutation_qvalue.txt") as f:
    header = f.readline()
    for line in f :
        fields = line.strip().split("\t")
        sJunction,snp,beta,npvalue,pvalue,qvalue = fields[0],fields[5],fields[8],fields[9],fields[10],fields[11]
        sJunc_SNP = sJunction+":"+snp
        npvalues.append(npvalues)
        bpvalues.append(bpvalues)
        if not qvalue=="NA" and float(qvalue) <= 0.05:
            if sJunction not in Joint_analysis_junctions:
                Joint_analysis_junctions.append(sJunction)
            Joint_analysis_SNpGenePairs_pval[sJunc_SNP] = -np.log10(float(pvalue))
            Joint_analysis_SNpGenePairs_beta[sJunc_SNP] = float(beta)

leaveOutHI_analysis_SNpGenePairs_pval = {}

    header = f.readline()
    for line in f :
        fields = line.strip().split(" ")
        sJunction,snp,beta,pvalue = fields[0],fields[1],fields[4],fields[3]
        sJunc_SNP = sJunction+":"+snp
        if sJunc_SNP in Joint_analysis_SNpGenePairs_pval:
            leaveOutHI_analysis_SNpGenePairs_pval[sJunc_SNP] = -np.log10(float(pvalue))
            leaveOutHI_analysis_SNpGenePairs_beta[sJunc_SNP] = float(beta)

y_val_pval = []
x_val_beta = []
y_val_beta = []

    y_val_pval.append(Joint_analysis_SNpGenePairs_pval[i])

    x_val_beta.append(leaveOutHI_analysis_SNpGenePairs_beta[i])
    y_val_beta.append(Joint_analysis_SNpGenePairs_beta[i])
